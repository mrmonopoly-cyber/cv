  \begin{itemize}
    \item Designed bare-metal firmware for Infineon AURIX TC375 (multi-core TriCore) without RTOS support
    \item Implemented lock-free and atomic synchronization primitives for inter-core communication
    \item Developed safety-critical fault handling using GPIO-based global emergency signaling
    \item Built deterministic Linux-based emulator (QEMU + SocketCAN) to validate runtime behavior without HIL hardware
    \item Implemented runtime trace stack for error diagnostics using \_\_FILE\_\_ and \_\_LINE\_\_
    \item \textbf{Repository:} \url{https://github.com/mrmonopoly-cyber/ControlUnitLogicOperator}
  \end{itemize}
}

\cventry{March 2026--Present}{\textbf{CRuntime}}{Freestanding Task Scheduler and Execution Runtime}{}{}{
  \begin{itemize}
    \item Designed a freestanding cooperative task scheduler for constrained environments (no OS / libc)
    \item Implemented hardware abstraction layer (HAL) for portability across bare-metal and Linux targets
    \item Developed lock-free data structures for task management and scheduling
    \item Implemented low-overhead task suspension leveraging CPU register state
    \item \textbf{Repository:} \url{https://github.com/mrmonopoly-cyber/CRuntime}
  \end{itemize}
}

\cventry{2024--2025}{\textbf{Debug Peripheral System (DPS)}}{Embedded Diagnostics and Calibration Framework}{RaceUp Formula SAE Team}{}{
  \begin{itemize}
    \item Designed CAN-based master/slave system for real-time diagnostics and live parameter calibration
    \item Developed command-line interface (CLI) for runtime interaction with embedded nodes
    \item Enabled remote inspection and tuning of firmware state without recompilation or redeployment
    \item \textbf{Repositories:} \url{https://github.com/mrmonopoly-cyber/DPS}, \url{https://github.com/mrmonopoly-cyber/dps_cli}
  \end{itemize}
}
